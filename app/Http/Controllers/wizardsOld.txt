    public function create($lesson_id = null, $currentWizard = null, $translation = null)

            }
        }

        $media_array = [];

                $activityAssignments = ActivityWizardAssignment::where('lesson_wizard_id', $added_step_record->id ?? '')->pluck('activity_id')->toArray();
            }
        }
        return view('admin.wizards.create', compact('stepLabel', 'stepLabelPrev', 'languages', 'activity_types', 'lesson_id', 'module_id', 'wizard', 'modules', 'unique_id', 'form_title', 'step_no', 'copy', 'main_wizard_id', 'activities', 'added_step_record', 'activityAssignments', 'media_array', 'translation', 'actual_content'));

    public function store(LessonWizardsRequest $request)
    {    
        $language_title = DB::table('languages')->where('language_id', $request->language_id)->value('language_title');
        $wizard = '';
        $label = $request->stepLabel;
        $stepLabelNext = $this->nextLabel[$label] ?? '';
        $is_translation = $request->translation ? true : false;
        
        if ($is_translation) {
            if ($request->stepLabel == 'step1') {
                $data = $this->saveLesson($request, $is_translation);
                $request->merge(['lesson_id' => $data->id]);
                $wizard = $this->saveWizards($is_translation, $request);
            } else {
                $wizard = $this->saveWizards($is_translation, $request);
            
               
                if ($label == 'step9') {
                    session()->flash('success', "Lesson wizard saved successfully!");
                    if ($request->copy == 1) {
                        return redirect()->route('create.lesson-wizard', [
                            'lesson_id' => $wizard->lesson_id,
                            'currentWizard' => 'step1',
                            'translation' => 'translation'
                        ]);
                    } else {
                        return redirect()->route('lessons-wizards');
                    }
                }
            }

            return redirect()->route('create.lesson-wizard', [
                'lesson_id' => $wizard->lesson_id,
                'currentWizard' => $stepLabelNext,
                'translation' => $language_title
            ]);
        } else { 
            if ($request->stepLabel == 'step1') {
                $data = $this->saveLesson($request, $is_translation);
                $request->merge(['lesson_id' => $data->id]);
                $wizard = $this->saveWizards($is_translation, $request);
            } else { 
                $wizard = $this->saveWizards($is_translation, $request);
            }
        }

        session()->flash('success', "Step {$request->step} saved successfully!");
        $label = $request->stepLabel;
        if ($label == 'step9') {
            session()->flash('success', "Lesson wizard saved successfully!");
            if ($request->copy == 1) {
                return redirect()->route('create.lesson-wizard', [
                    'lesson_id' => $wizard->lesson_id,
                    'currentWizard' => 'step1',
                    'translation' => 'translation'
                ]);
            } else {
                return redirect()->route('lessons-wizards');
            }
        } else {
            $stepLabelNext = $this->nextLabel[$label] ?? '';
            return redirect()->route('create.lesson-wizard', [
                'lesson_id' => $wizard->lesson_id,
                'currentWizard' => $stepLabelNext,
            ]);
        }
    }
    public function saveLesson($request, $is_translation)
    {    
        $model = $is_translation ? "App\\Models\\LessonsTranslationsModel" : "App\\Models\\LessonsModel";
        $data = $request->lesson_id ? $model::find($request->lesson_id)->where('language_id',$request->language_id)->first() : new $model(); 
     
        if(!$data){
        $data =new $model();
        }
        $data->title = $request->title;
        $this->assignCommonFields($data, $request);
        if ($is_translation) {
            $data->main_lesson_id = $request->lesson_id;
        } else {
            $data->slug = $request->slug;
            $data->module_id = $request->lesson_module;
        }
         
        $data->save();
        if (!$is_translation) {
            $this->saveLessonTranslation($data, $request);
        }
        return $data;
    }

    public function saveWizards($is_translation, $request)
    {
        // dd($request->step_unique_id);
        $created_at = Carbon::now();
        $created_by = Auth::user()->id;

        $model              =  $is_translation ? "App\\Models\\LessonWizardTranslations" : "App\\Models\\LessonsWizardsModel";
        $wizard             =  $request->wizard_id ? $model::find($request->wizard_id) : new $model();
       
        if ($request->lesson_module) {
          
            $module                  =     LessonModules::find($wizard->module_id);
            $is_sensitive            =     $module ? $module->value('is_sensitive') : 0; 
            $wizard->is_sensitive    =     $is_sensitive;
        }
        if ($is_translation) {
          
            $wizard->main_wizard_id = $request->main_wizard_id;
        }else{
            $wizard->module_id  =  $request->lesson_module ?? $request->module_id ?? '';
        }
        $wizard->lesson_id          =   $request->lesson_id;
        $wizard->title              =   $request->title;
        $wizard->text               =   $request->text;
        $wizard->media_type         =   $request->TransationMediaType ?? $request->media_type ?? 0;
        $wizard->learning_points    =   $request->learning_points;
        $wizard->description        =   $request->description;
        $wizard->wizard             =   $request->wizard;
        $this->assignCommonFields($wizard, $request);
        $wizard->save();
        if(!$is_translation){
            $this->saveWizardTranslations($is_translation,$request,$wizard);
        }
        if ($request->step == 6 || $request->step == 4) {
            DB::table('activity_wizard_assignments')->where('lesson_wizard_id', $wizard->id)->where('language_id', $wizard->language_id)->delete();
            foreach ($request->activityType as $key => $value) {
                $assignment                     =   new ActivityWizardAssignment();
                $assignment->lesson_wizard_id   =   $wizard->id;
                $assignment->language_id        =   $wizard->language_id;
                $assignment->activity_id        =   $value;
                $assignment->created_at         =   $created_at;
                $assignment->created_by         =   $created_by;
                $assignment->updated_by         =   $created_by;
                $assignment->updated_at         =   $created_at;
                $assignment->save();
            }
        }
        LessonWizardsMediaModel::where('unique_id', $request->step_unique_id)->update([
            'content_id' => $wizard->id,
            'model'      => $request->wizard,
            'lesson_id'  => $wizard->lesson_id
        ]);

        if ($request->hasFile('image_transaction')) {
            $completeFileName   =   $request->file('image_transaction')->getClientOriginalName();
            $fileNameOnly       =   pathinfo($completeFileName, PATHINFO_FILENAME);
            $extension          =   $request->file('image_transaction')->getClientOriginalExtension();
            $image              =   str_replace(' ', '_', $fileNameOnly) . '.' . $extension;
            $request->file('image_transaction')->storeAs('public/lesson/images', $image);

            $image              =   'storage/lesson/images/' . $image;
            $LessonWizardsMediaModel = LessonWizardsMediaModel::where('lesson_id', $wizard->lesson_id)->where('model', $request->wizard)->where('content_id', $wizard->id)->first();
            if (!$LessonWizardsMediaModel) {
                $LessonWizardsMediaModel = new LessonWizardsMediaModel();
            }
            $LessonWizardsMediaModel->image_source =   $image;
            $LessonWizardsMediaModel->lesson_id    =   $wizard->lesson_id;
            $LessonWizardsMediaModel->model        =   $wizard->wizard;
            $LessonWizardsMediaModel->content_id   =   $wizard->id;
            $LessonWizardsMediaModel->created_at   =   $created_by;
            $LessonWizardsMediaModel->created_by   =   $created_at;
            $LessonWizardsMediaModel->save();
        }
        // else {
        //     if ($wizard->media_type != 2) {
        //         LessonWizardsMediaModel::where('lesson_id', $wizard->lesson_id)->where('model',  $request->wizard)->where('content_id', $wizard->id)->delete();
        //     }
        // }

        return $wizard;
    }

    protected function assignCommonFields($data, $request)
    {

        if ($request->wizard_id) {
            $data->updated_at   = Carbon::now();
            $data->updated_by   = Auth::user()->id;
        } else {
            $data->created_at   = Carbon::now();
            $data->created_by   = Auth::user()->id;
            $data->language_id  = $request->language_id;
            $data->updated_at   = null;
            $data->updated_by   = null;
        }
    }

    protected function assignNonTranslationFields($activity, $request)

        $activity->content_type  = $request->activity_content_type == 0 ? 1 : $request->activity_content_type;
        $activity->is_true       = $request->true_false;
        $activity->activity_type = $request->activity_type;
        $activity->no_of_attempts= $request->no_of_attempts;
        $activity->show_feedback = $request->show_feedback;
        $activity->time_limit    = $request->duration ?? 0;
    }

        $translation->fill($translationAttributes);

    public function saveActivityMedia($activity, $request)
    {
        if (isset($request->activityImages) && is_array($request->activityImages)) {
            $existingMediaIds = [];
            // Process each activity image data
            foreach ($request->activityImages as $index => $data) {
                $activityMedia = null;
                if (isset($data['media_id']) && !empty($data['media_id'])) {
                    // Update existing media
                    $activityMedia = ActivityMedia::where('id', $data['media_id'])
                        ->where('model', 'activity')
                        ->where('content_id', $activity->id)
                        ->first();
                }
                if (!$activityMedia) {
                    // Create new media entry
                    $activityMedia = new ActivityMedia();
                    $activityMedia->created_by = Auth::user()->id;
                    $activityMedia->model = 'activity';
                    $activityMedia->content_id = $activity->id;
                    $activityMedia->language_id = $activity->language_id;
                } else {
                    // Update existing media entry
                    $activityMedia->updated_by = Auth::user()->id;
                }
                if (!empty($data['file']) && $data['file'] instanceof \Illuminate\Http\UploadedFile) {
                    $file = $data['file'];
                    $completeFileName = $file->getClientOriginalName();
                    $fileNameOnly = pathinfo($completeFileName, PATHINFO_FILENAME);
                    $extension = $file->getClientOriginalExtension();
                    $empPicture = str_replace(' ', '_', $fileNameOnly) . '_' . time() . '.' . $extension;
                    $file->storeAs('public/activity-images', $empPicture);
                    $image = '/storage/activity-images/' . $empPicture;
                    $activityMedia->image_source = $image;
                } elseif (!empty($data['hiddenFile'])) {
                    // Handle hidden file
                    $activityMedia->image_source = $data['hiddenFile'];
                }
                $activityMedia->save();
                $existingMediaIds[] = $activityMedia->id;
            }
            ActivityMedia::where('model', 'activity')
                ->where('content_id', $activity->id)
                ->whereNotIn('id', $existingMediaIds)
                ->delete();
        }
    }

    public function destroy($id)
    {
        $activity = ActivitiesModel::with('translations')->findOrFail($id);
        foreach ($activity->translations as $translation) {
            $translation->update([
                'deleted_at' => now(),
                'deleted_by' => auth()->id(),
            ]);
            $optionIds = ActivityOptions::where('activity_id', $translation->main_activity_id)->pluck('id')->toArray();

            ActivityOptions::whereIn('id', $optionIds)->update([
                'deleted_at' => now(),
                'deleted_by' => auth()->id(),
            ]);
            ActivitiesOptionsTranslations::whereIn('main_option_id', $optionIds)->update([
                'deleted_at' => now(),
                'deleted_by' => auth()->id(),
            ]);
            ActivityMedia::where('model', 'option')
                ->whereIn('content_id', $optionIds)
                ->update([
                    'deleted_at' => now(),
                    'deleted_by' => auth()->id(),
                ]);
        }
        ActivityMedia::where('model', 'activity')
            ->where('content_id', $id)
            ->update([
                'deleted_at' => now(),
                'deleted_by' => auth()->id(),
            ]);
        $activity->update([
            'deleted_at' => now(),
            'deleted_by' => auth()->id(),
        ]);

        return response()->json(['status' => 'success', 'msg' => 'Activity Deleted Successfully!']);
    }


    public function validateWizardVideos(Request $request)
    {

        $exisiting_videos = [];
        $video_to_be_replace = [];
        $requestedFiles = $request->input('files');
        $msg = 'All Videos are unique!';
        $status = 'success';
        foreach ($requestedFiles as $file) {
            $fileInfo = extractFilebyPart($file);
            $is_video = '';
            $query = LessonWizardsMediaModel::where('video_view', $fileInfo['view_text'])->where('language_id', $request->language_id)->where('video_quality', $fileInfo['resolution']);
            if ($request->wizard_id) {  //FOR  Edit
                $is_video = $query->where('lesson_id', $request->lesson_id)
                    ->where('content_id', $request->wizard_id)
                    ->first();
            } else if ($request->step_unique_id) {   //FOR  CREATE
                $is_video = $query->where('content_id', '=', null)
                    ->where('unique_id', $request->step_unique_id)
                    ->first();
            }
            if ($is_video) {
                $exisiting_videos[] = $file;
                $video_to_be_replace[] = $is_video->file_name;
                $msg = 'Video with same quailty already Exist';
                $status = 'exisit';
            }
        }
        return response()->json(['status' => $status, 'files' => $exisiting_videos, 'videos_to_be_replace' => $video_to_be_replace, 'message' => $msg]);
    }
    public function deleteWizardVideo(Request $request, $id)
    {
        $query = LessonWizardsMediaModel::where('model', $request->model);
        if ($id != 'null' && $id > 0 && $id != 'undefined') {
            $query = $query->where('content_id', $id)->first();
        } else if ($request->unid) {
            $query = $query->where('unique_id', $request->unid)->first();
        }
        if ($query) {
            Storage::disk('spaces')->delete($query->video_source);
            $query->delete();
            return response()->json([
                'status' => 'success'
            ]);
        }
        return response()->json([
            'status' => 'failed'
        ]);
    }
